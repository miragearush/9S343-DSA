/**
 * B.Ropes
 * Link : https://codeforces.com/edu/course/2/lesson/6/2/practice/contest/283932/problem/B
 * Topic : Binary Search on Answers (Decimal variant)
 */
#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

bool possible(double mid,double a[],int n,int k)
{
    int done=0;
    for(int i=0;i<n;i++)
    {
        done+=(a[i]/mid);
    }
    return done>=k;
}

int main(){
    int n,k;
    cin>>n>>k;
    double a[n];
    for(int i=0;i<n;i++)
        cin>>a[i];
    double low=0,high=1e18,ans=0;
    double precision=1e-7;
    while(high-low>precision)
    {
        double mid=low+((high-low)/2);
        if(possible(mid,a,n,k))
        {
            ans=mid;
            low=mid;
        }
        else
            high=mid;
    }
    cout<<fixed<<setprecision(7);
    cout<<ans;
    return 0;
}
