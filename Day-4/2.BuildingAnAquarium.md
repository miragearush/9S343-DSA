/**
 * E.Building An Aquarium
 * Link : https://codeforces.com/contest/1873/problem/E
 * Topic : Binary Search on Answers
 */
#include <bits/stdc++.h>
using namespace std;

bool check(int height,vector<int>&a,int w) {
    long long waterReq=0;

    for(int i=0;i<a.size();i++) {
        if(height>a[i]) {
            waterReq+=(0ll+height-a[i]);
        }
        if(waterReq>w)
            return false;
    }
    return true;
}

int main() 
{
    //inputs
    int t;
    cin>>t;
    while(t--) {
        int n,w;
        cin>>n>>w;
        vector<int>a(n);
        for(int i=0;i<n;i++) {
            cin>>a[i];
        }
        
        long long low=1,high=(*max_element(a.begin(),a.end())+w);
        long long ans=0;
        while(low<=high) {
            long long mid = low+(high-low)/2;

            if(check(mid,a,w)==true) {
                ans=mid;
                low=mid+1;
            }
            else    
                high=mid-1;
        }
        cout<<ans<<endl;
    }
    return 0;
}
