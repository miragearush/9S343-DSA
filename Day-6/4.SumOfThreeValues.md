/**
* Sum Of Three Values
* Link : https://cses.fi/problemset/task/1641/
* Topic : Two Pointers
*/
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, x;
    cin >> n >> x;

    vector<pair<int, int>> a(n);

    for (int i = 0; i < n; i++) {
        cin >> a[i].first;
        a[i].second = i + 1;
    }

    sort(a.begin(), a.end());

    bool found = false;

    for (int i = 0; i < n - 2; i++) {
        int l = i + 1;
        int r = n - 1;

        while (l < r) {
            long long sum = 1LL * a[i].first + a[l].first + a[r].first;

            if (sum == x) {
                cout << a[i].second << " "
                     << a[l].second << " "
                     << a[r].second;
                found = true;
                break;
            }

            if (sum < x)
                l++;
            else
                r--;
        }

        if (found)
            break;
    }

    if (!found)
        cout << "IMPOSSIBLE";
}
